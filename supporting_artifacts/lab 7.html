rescale_01 <- function(x){
  stopifnot(is.numeric(x) | length(x) <= 1)
  
  my_max = max(x, na.rm = TRUE)
  my_min = min(x, na.rm = TRUE)
  range = my_max - my_min
  rescale = (x - my_min)/range
  return(rescale)
}

rescale_column <- function(df, variables){
  stopifnot(is.data.frame(df))
  new_df <- df |> 
    mutate(
      across(
        .cols = {{variables}},
        .fns = rescale_01)
    )
  return(new_df)
}
