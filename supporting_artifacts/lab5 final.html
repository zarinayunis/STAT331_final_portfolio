surveys |> 
  select(weight, genus, year) |> 
  group_by(genus, year) |> 
  summarize(weight = mean(weight)) |> 
  ggplot(mapping = aes(y = weight, 
                       x = year, 
                       col = fct_reorder2(genus, year, weight, .desc = TRUE))) +
  geom_line() +
  labs(title = "Reordering Genus by Mean Weight", 
       y = "Mean Weight", 
       col = "Color by Genus")
